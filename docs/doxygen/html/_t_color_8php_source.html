01121     private function _tokenize($string) {
01122         $tokens = array();
01123         $last = 0;
01124         for ($i = 0; $i < strlen($string); $i++) {
01125             $part = substr($string, $i, 2);
01126             if (in_array($part, self::$FORMATS)) {
01127                 $text = substr($string, $last, $i - $last);
01128                 if (!empty($text) && !in_array($text, self::$FORMATS))
01129                     $tokens[] = $text;
01130                 $last = $i + 2;
01131                 $tokens[] = $part;
01132             }
01133         }
01134         $text = substr($string, $last, strlen($string) - $last);
01135         if (!empty($text) && !in_array($text, self::$FORMATS))
01136             $tokens[] = $text;
01137         return $tokens;
01138     }
